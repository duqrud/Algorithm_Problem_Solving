import sys
sys.setrecursionlimit(10**9)

N = int(input())
node = []
visited = []
result = []

for i in range(N+1):
    node.append([])
    visited.append(0)
    result.append(0)

for i in range(N-1):
    x,y = map(int, input().split())
    node[x].append(y)
    node[y].append(x)

def dfs(n):
    for i in node[n]:
        if not visited[i]:
            visited[i] = 1
            result[i] = n
            dfs(i)
visited[1] = 1
dfs(1)

for i in result[2:]:
    print(i)
