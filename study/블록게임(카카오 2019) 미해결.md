def solution(board):
    answer = 0
    visit = [[0 for _ in range(len(board))] for i in range(len(board))]
    while True:
        answer2 = answer
        stack = [0 for _ in range(len(board))]
        for i in range(len(board)):
            for j in range(len(board[i])):
                if board[i][j] != 0:
                    if stack[j] == 0: 
                        stack[j] = 1
                        try:
                            if board[i-1][j] == 0:
                                board[i-1][j] = -1
                                visit[i-1][j] = 1
                        except:
                            continue
                        try:
                            if board[i-2][j] == 0:
                                board[i-2][j] = -1
                                visit[i-2][j] = 1
                        except:
                            continue
        for i in range(len(board)):
            for j in range(len(board[i])):
                if board[i][j] != 0:
                    stack2 = []
                    for x in range(2):
                        for y in range(3):
                            try:
                                if board[i+x][j+y] != 0:
                                    stack2.append(board[i+x][j+y])
                            except:
                                continue
                    if len(stack2) == 6:
                        if len(list(filter(lambda a: a != -1, stack2))) == 4:
                            if len(set(stack2)) == 2:
                                answer += 1
                                for x in range(2):
                                    for y in range(3):
                                        board[i+x][j+y] = 0
                                break
                    stack2 = []
                    for x in range(2):
                        for y in range(3):
                            try:
                                if board[i+x][j+y] != 0:
                                    stack2.append(board[i+y][j+x])
                            except:
                                continue
                    if len(stack2) == 6:
                        if len(list(filter(lambda a: a != -1, stack2))) == 4:
                            if len(set(stack2)) == 2:
                                answer += 1
                                for x in range(2):
                                    for y in range(3):
                                        board[i+y][j+x] = 0
        for i in range(len(board)):
            for j in range(len(board[i])):
                if visit[i][j] == 1:
                    board[i][j] = 0
                    visit[i][j] = 0
        if answer2 == answer:
            break
    return answer
